import asyncio

async def validate_environment():

        'AZURE_CLIENT_ID',
        'AZURE_CLIENT_SECRET',
        'AZURE_TENANT_ID'

    for var in required_env_vars:
        value = os.getenv(var)
        results[f'env_{var}'] = {

        }

    try:
        conn = await asyncpg.connect(

            port=int(os.getenv('POSTGRES_PORT', 5432)),

            'status': 'fail',
            'error': str(e)
